--oracle database

DROP TABLE STUDENTCLASS;
DROP TABLE RESULT;
DROP TABLE STUDENT;
DROP TABLE CLASS;
DROP TABLE COURSES;


CREATE TABLE STUDENT(
STUDENT_NO CHAR(8),
NAME VARCHAR(20),
AGE INTEGER);

CREATE TABLE CLASS(
CLASS_NO CHAR(6),
TEACHER_NAME VARCHAR(20));

CREATE TABLE STUDENTCLASS(
STUDENT_NO CHAR(8),
CLASS_NO CHAR(6));

CREATE TABLE COURSES(
COURSES_NO CHAR(2),
NAME VARCHAR(40),
CREDIT INTEGER);

CREATE TABLE RESULT(
STUDENT_NO CHAR(8),
COURSES_NO CHAR(2),
SCORE INTEGER);
